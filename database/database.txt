CREATE TABLE IF NOT EXISTS student_inquiries (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    full_name       VARCHAR(100)    NOT NULL,
    email           VARCHAR(150)    NOT NULL,
    phone           VARCHAR(20)     NOT NULL,
    current_edu     VARCHAR(50)     NOT NULL,
    preferred_country VARCHAR(50)   NOT NULL,
    preferred_faculty VARCHAR(50)   NOT NULL,
    study_level     VARCHAR(20)     NOT NULL,
    english_score   VARCHAR(50),
    budget          VARCHAR(30),
    message         TEXT,
    submitted_at    DATETIME        DEFAULT CURRENT_TIMESTAMP
);
